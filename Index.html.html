// ============ ADD TRANSACTION ============
function addTransaction() {
const amount = parseFloat(document.getElementById('amount').value);
const date = document.getElementById('txnDate').value;
const description = document.getElementById('description').value.trim();
const category = document.getElementById('category').value;
const paymentMode = document.getElementById('paymentMode').value;
const notes = document.getElementById('notes').value.trim();
const tags = document.getElementById('tags').value.trim();
const merchant = document.getElementById('merchant').value.trim();

if (!amount || amount <= 0) return showToast('⚠️ Enter a valid amount');
if (!description) return showToast('⚠️ Enter a description');
if (!date) return showToast('⚠️ Select a date');
if (!paymentMode) return showToast('⚠️ Select payment mode');

const txn = {
id: Date.now(),
amount, date, description, category,
paymentMode, notes, merchant,
tags: tags ? tags.split(',').map(t => t.trim()).filter(Boolean) : [],
type: currentType,
createdAt: new Date().toISOString()
};

transactions.unshift(txn);
saveTxns();
renderAll();
clearForm();
showToast('✅ Transaction added!');
}

function updateHeader() {
const now = new Date();
const monthTxns = transactions.filter(t => {
const d = new Date(t.date);
return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();
});
const income = monthTxns.filter(t => t.type === 'income').reduce((s, t) => s + t.amount, 0);
const expense = monthTxns.filter(t => t.type === 'expense').reduce((s, t) => s + t.amount, 0);
const net = income - expense;

document.getElementById('totalIncome').textContent = '₹' + fmt(income);
document.getElementById('totalExpense').textContent = '₹' + fmt(expense);
document.getElementById('txnCount').textContent = monthTxns.length;
const balEl = document.getElementById('netBalance');
balEl.textContent = (net < 0 ? '-₹' : '₹') + fmt(Math.abs(net));
balEl.className = 'balance-amount ' + (net < 0 ? 'negative' : 'positive');
}

const recent = transactions.slice(0, 8);
const el = document.getElementById('recentList');
if (!recent.length) {
el.innerHTML = `<div class="empty-state"><div class="emoji">💳</div><h3>No transactions yet</h3><p>Add your first transaction above</p></div>`;
return;
}
el.innerHTML = recent.map(t => txnHTML(t)).join('');
}

function renderTransactions() {
const search = (document.getElementById('searchInput')?.value || '').toLowerCase();
const filtered = transactions.filter(t => {
const d = new Date(t.date);
const monthMatch = d.getMonth() === currentMonth && d.getFullYear() === currentYear;
const filterMatch = currentFilter === 'all' || t.type === currentFilter || t.category === currentFilter;
const searchMatch = !search || t.description.toLowerCase().includes(search) ||
t.category.toLowerCase().includes(search) || t.paymentMode.toLowerCase().includes(search) ||
(t.merchant && t.merchant.toLowerCase().includes(search)) ||
(t.notes && t.notes.toLowerCase().includes(search));
return monthMatch && filterMatch && searchMatch;
});

const el = document.getElementById('txnList');
if (!filtered.length) {
el.innerHTML = `<div class="empty-state"><div class="emoji">🔍</div><h3>No transactions</h3><p>Try a different filter or month</p></div>`;
return;
}

// Group by date
const grouped = {};
filtered.forEach(t => {
if (!grouped[t.date]) grouped[t.date] = [];
grouped[t.date].push(t);
});

el.innerHTML = Object.keys(grouped).sort((a,b) => b.localeCompare(a)).map(date => {
const dayTotal = grouped[date].reduce((s,t) => t.type==='expense'?s-t.amount:s+t.amount, 0);
const dateLabel = new Date(date + 'T00:00:00').toLocaleDateString('en-IN', { weekday:'short', day:'numeric', month:'short' });
return `
<div style="margin-bottom:4px">
<div style="display:flex;justify-content:space-between;align-items:center;padding:8px 0 4px;border-bottom:1px solid var(--border)">
<span style="font-size:12px;font-weight:700;color:var(--text2);text-transform:uppercase;letter-spacing:0.5px">${dateLabel}</span>
<span style="font-size:12px;font-weight:700;color:${dayTotal>=0?'var(--green)':'var(--red)'}">${dayTotal>=0?'+':''} ₹${fmt(Math.abs(dayTotal))}</span>
</div>
${grouped[date].map(t => txnHTML(t)).join('')}
</div>`;
}).join('');
}

// ============ ANALYTICS ============
function renderAnalytics() {
const now = new Date();
const monthTxns = transactions.filter(t => {
const d = new Date(t.date);
return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear() && t.type === 'expense';
});
const incomeTxns = transactions.filter(t => {
const d = new Date(t.date);
return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear() && t.type === 'income';
});

const totalExp = monthTxns.reduce((s,t) => s+t.amount, 0);
const totalInc = incomeTxns.reduce((s,t) => s+t.amount, 0);
const daysInMonth = new Date(now.getFullYear(), now.getMonth()+1, 0).getDate();
const avgDaily = totalExp / (now.getDate());

// Analytics cards
document.getElementById('avgDaily').textContent = '₹' + fmt(avgDaily);
document.getElementById('savingsRate').textContent = totalInc > 0 ? Math.round((1 - totalExp/totalInc) * 100) + '%' : '—';

// Top category
const catTotals = {};
monthTxns.forEach(t => { catTotals[t.category] = (catTotals[t.category]||0) + t.amount; });
const topCat = Object.keys(catTotals).sort((a,b) => catTotals[b]-catTotals[a])[0];
document.getElementById('topCategory').textContent = topCat ? categoryEmoji(topCat) + ' ' + topCat.split(' ')[0] : '—';

// Top payment
const pmTotals = {};
monthTxns.forEach(t => { pmTotals[t.paymentMode] = (pmTotals[t.paymentMode]||0) + t.amount; });
const topPm = Object.keys(pmTotals).sort((a,b) => pmTotals[b]-pmTotals[a])[0];
document.getElementById('topPayment').textContent = topPm ? pmIconFor(topPm) + ' ' + (topPm.length > 8 ? topPm.slice(0,8)+'…' : topPm) : '—';

// Day chart (last 7 days)
const chartEl = document.getElementById('dayChart');
const days = [];
for (let i = 6; i >= 0; i--) {
const d = new Date(); d.setDate(d.getDate() - i);
const dateStr = d.toISOString().split('T')[0];
const dayExp = transactions.filter(t => t.date === dateStr && t.type === 'expense').reduce((s,t) => s+t.amount, 0);
days.push({ label: d.toLocaleDateString('en-IN', { weekday: 'short' }).slice(0,2), amount: dayExp });
}
const maxAmt = Math.max(...days.map(d => d.amount), 1);
chartEl.innerHTML = days.map(d => {
const pct = (d.amount / maxAmt) * 100;
return `<div class="chart-bar-col">
<div class="chart-bar" style="height:${Math.max(pct,4)}%"></div>
<div class="chart-label">${d.label}</div>
</div>`;
}).join('');

// Category breakdown
const catEl = document.getElementById('categoryBreakdown');
if (!Object.keys(catTotals).length) { catEl.innerHTML = '<div style="color:var(--text2);font-size:13px;text-align:center;padding:16px 0">No expense data this month</div>'; }
else {
const maxCat = Math.max(...Object.values(catTotals));
catEl.innerHTML = Object.keys(catTotals).sort((a,b) => catTotals[b]-catTotals[a]).map(cat => `
<div class="cat-item">
<div class="cat-dot" style="background:${CATEGORY_COLORS[cat]||'#8888aa'}"></div>
<div class="cat-name">${categoryEmoji(cat)} ${cat}</div>
<div class="cat-bar-wrap"><div class="cat-bar-fill" style="width:${(catTotals[cat]/maxCat)*100}%;background:${CATEGORY_COLORS[cat]||'#8888aa'}"></div></div>
<div class="cat-amount" style="color:${CATEGORY_COLORS[cat]||'#8888aa'}">₹${fmt(catTotals[cat])}</div>
</div>`).join('');
}

// Payment breakdown
const pmEl = document.getElementById('paymentBreakdown');
if (!Object.keys(pmTotals).length) { pmEl.innerHTML = '<div style="color:var(--text2);font-size:13px;text-align:center;padding:16px 0">No expense data this month</div>'; }
else {
const maxPm = Math.max(...Object.values(pmTotals));
pmEl.innerHTML = Object.keys(pmTotals).sort((a,b) => pmTotals[b]-pmTotals[a]).map(pm => `
<div class="cat-item">
<div style="font-size:18px;width:24px">${pmIconFor(pm)}</div>
<div class="cat-name">${pm}</div>
<div class="cat-bar-wrap"><div class="cat-bar-fill" style="width:${(pmTotals[pm]/maxPm)*100}%;background:var(--accent)"></div></div>
<div class="cat-amount" style="color:var(--accent)">₹${fmt(pmTotals[pm])}</div>
</div>`).join('');
}
}

budgets[cat] = amt;
localStorage.setItem('spendly_budgets', JSON.stringify(budgets));
renderBudget();
showToast('✅ Budget set!');
document.getElementById('budgetAmount').value = '';
}

function renderBudget() {
const now = new Date();
const monthExpenses = transactions.filter(t => {
const d = new Date(t.date);
return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear() && t.type === 'expense';
});
const catTotals = {};
monthExpenses.forEach(t => { catTotals[t.category] = (catTotals[t.category]||0) + t.amount; });

const el = document.getElementById('budgetList');
if (!Object.keys(budgets).length) {
el.innerHTML = `<div class="empty-state" style="padding:24px 0"><div class="emoji" style="font-size:32px">🎯</div><p style="font-size:13px">No budgets set yet</p></div>`;
return;
}
el.innerHTML = Object.keys(budgets).map(cat => {
const limit = budgets[cat];
const spent = catTotals[cat] || 0;
const pct = Math.min((spent/limit)*100, 100);
const cls = pct >= 100 ? 'over' : pct >= 80 ? 'warn' : 'ok';
const status = pct >= 100 ? '⚠️ Over budget!' : pct >= 80 ? '⚡ Almost there' : '✅ On track';
return `
<div class="budget-item">
<div class="budget-top">
<div class="budget-cat">${categoryEmoji(cat)} ${cat}</div>
<div class="budget-vals">₹${fmt(spent)} / ₹${fmt(limit)}</div>
</div>
<div class="budget-progress"><div class="budget-fill ${cls}" style="width:${pct}%"></div></div>
<div style="font-size:11px;color:var(--text2);margin-top:4px;display:flex;justify-content:space-between">
<span>${status}</span>
<span style="cursor:pointer;color:var(--red)" onclick="deleteBudget('${cat}')">Remove</span>
</div>
</div>`;
}).join('');
}

function deleteBudget(cat) {
delete budgets[cat];
localStorage.setItem('spendly_budgets', JSON.stringify(budgets));
renderBudget();
}

function downloadExcel() {
if (!transactions.length) return showToast('⚠️ No transactions to export');
const data = transactions.map(t => ({
Date: t.date, Type: t.type, Amount: t.amount, Description: t.description,
Category: t.category, 'Payment Mode': t.paymentMode, Merchant: t.merchant || '',
Notes: t.notes || '', Tags: t.tags.join(', ')
}));
const ws = XLSX.utils.json_to_sheet(data);
const wb = XLSX.utils.book_new();
XLSX.utils.book_append_sheet(wb, ws, 'Transactions');

// Summary sheet
const now = new Date();
const monthTxns = transactions.filter(t => { const d = new Date(t.date); return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear(); });
const income = monthTxns.filter(t=>t.type==='income').reduce((s,t)=>s+t.amount,0);
const expense = monthTxns.filter(t=>t.type==='expense').reduce((s,t)=>s+t.amount,0);
const summary = [
{ Metric: 'Total Income (This Month)', Value: income },
{ Metric: 'Total Expense (This Month)', Value: expense },
{ Metric: 'Net Balance', Value: income - expense },
{ Metric: 'Total Transactions', Value: transactions.length },
{ Metric: 'Export Date', Value: new Date().toLocaleDateString('en-IN') }
];
const ws2 = XLSX.utils.json_to_sheet(summary);
XLSX.utils.book_append_sheet(wb, ws2, 'Summary');
XLSX.writeFile(wb, `Spendly_Export_${new Date().toISOString().split('T')[0]}.xlsx`);
showToast('📊 Excel downloaded!');
}

function downloadCSV() {
if (!transactions.length) return showToast('⚠️ No data to export');
const headers = ['Date','Type','Amount','Description','Category','Payment Mode','Merchant','Notes','Tags'];
const rows = transactions.map(t => [
t.date, t.type, t.amount, `"${t.description}"`, t.category, t.paymentMode,
t.merchant || '', `"${t.notes || ''}"`, t.tags.join(';')
]);
const csv = [headers, ...rows].map(r => r.join(',')).join('\n');
const blob = new Blob([csv], { type: 'text/csv' });
const a = document.createElement('a');
a.href = URL.createObjectURL(blob);
a.download = `Spendly_${new Date().toISOString().split('T')[0]}.csv`;
a.click();
showToast('📄 CSV downloaded!');
}

function fmt(n) {
if (n >= 1e7) return (n/1e7).toFixed(1) + 'Cr';
if (n >= 1e5) return (n/1e5).toFixed(1) + 'L';
if (n >= 1e3) return (n/1e3).toFixed(1) + 'K';
return Number(n).toFixed(0);
}

'Salary': '💼', 'Freelance': '🖥️', 'Business': '🏢', 'Other': '📌'

}
